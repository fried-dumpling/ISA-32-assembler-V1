#define IO_BASE 0x4000000
#define KEY_IN 0
#define KEY_V 4
#define TERM_OUT 8

#macro HALT 0
##inf_loop:
    jmp.one zero.32B, ##inf_loop
#end

#define buffSize 20

.data
    strBuff: 4*buffSize, 0   ; input buffer
    numA:   4, 0    ; inputA
    numB:   4, 0    ; inputB
    numOut: 4, 0    ; output
    4*20, 0
    stackEnd:
.text
    ; setting gen[24] to IO_BASE
    set gen[24].S16H, IO_BASE >> 16
    set gen[24].16L, IO_BASE & 0xFFFF

    set stack.32B, stackEnd     ; stack = stackEnd
    
    ; get numA
    call zero.32B, func_getInput
    call zero.32B, func_convertInt
    st.zero gen[0].32B, numA

    ; get numB
    call zero.32B, func_getInput
    call zero.32B, func_convertInt
    st.zero gen[0].32B, numB

    ; mem[numOut] = mem[numA] + mem[numB]
    ld.zero gen[1].32B, numA
    ld.zero gen[2].32B, numB
    add gen[1].32B, gen[2].32B
    st.zero gen[1].32B, numOut


    ; print ans
    ld.zero gen[1].32B, numOut
    call zero.32B, func_convertStr
    call zero.32B, func_print

    HALT()

    ; <get input>
    ; out: mem[strBuff]
    ; use: gen[1] -> counter
    ; use: gen[2] -> strBuff base
    ; use: gen[3] -> valid holder
    ; use: gen[4] -> char holder
func_getInput:
    set gen[1].32B, 1           ; gen[1](counter) = 1
    set gen[2].32B, strBuff    ; gen[2](base) = strBuff
    set gen[3].32B, 0           ; gen[3] = 0

getInput_mainloop:
    ; check valid
    ld.gen[24] gen[3].32B, KEY_V        ; gen[3](valid) = mem[IO_BASE + KEY_V]
    cmp gen[3].32B, zero.32B            ; valid == 0
    jmp.zero zero.32B, getInput_skip    ; if !valid skip

    ; get input
    ld.gen[24] gen[4].32B, KEY_IN       ; char = mem[IO_BASE + KEY_IN]
    st.gen[24] gen[4].32B, TERM_OUT     ; mem[IO_BASE + TERM_OUT] = char
    cmpi gen[4].32B, 0x0A               ; char == '\n'
    jmp.zero zero.32B, getInput_exit    ; if (char == '\n') break

    st.gen[2] gen[4].32B, 0     ; mem[base] = char
    addi gen[2].32B, 4          ; base += 4
    addi gen[1].32B, 1          ; count++
    cmpi gen[1].32B, buffSize   ; count == buffSize
    jmp.zero zero.32B, getInput_exit    ; if (count == buffSize) break
    
getInput_skip:
    jmp.one zero.32B, getInput_mainloop ; continue to loop

getInput_exit:
    set gen[4].32B, 0       ; char = 0
    st.gen[2] gen[4].32B, 0 ; mem[base] = char(0)
    ret

    ; <convert int>
    ; in: mem[strBuff]
    ; out: gen[0] -> number
    ; use: gen[1] -> char
    ; use: gen[2] -> base
    ; use: gen[3] -> tmp
    ; use: gen[4] -> const 10
func_convertInt:
    set gen[0].32B, 0           ; number = 0
    set gen[2].32B, strBuff    ; base = strBuff
    
convertInt_mainloop:
    ld.gen[2] gen[1].32B, 0     ; char = mem[base]
    cmp gen[1].32B, zero.32B    ; char == 0
    jmp.zero zero.32B, convertInt_exit  ; if (char == 0) break

    ;number *= 10
    mov gen[3].32B, gen[0].32B      ; gen[3] = gen[0]
    shiftli gen[0].32B, 1           ; gen[0] * 2
    shiftli gen[3].32B, 3           ; gen[3] * 8
    add gen[0].32B, gen[3].32B      ; gen[0] += gen[3]

    addi gen[2].32B, 4          ; base += 4
    subi gen[1].32B, 48         ; char -= 48
    add gen[0].32B, gen[1].32B  ; number += char
    jmp.one zero.32B, convertInt_mainloop   ; continue loop
    
convertInt_exit:
    ret

    ; <convert str>
    ; in: gen[1] -> number
    ; out: mem[strBuff] -> string
    ; use: gen[2] -> base
    ; use: gen[3] -> result
    ; use: gen[4] -> offset
    ; use: gen[5] -> tmp
    ; use: gen[6] -> const 3
    ; use: gen[7] -> counter
    ; use: gen[8] -> sum
func_convertStr:
    set gen[2].32B, strBuff ; base = strBuff
    set gen[3].32B, 0       ; result = 0
    set gen[7].32B, 0       ; counter = 0

convertStr_mainloop:
    shiftli gen[1].32B, 1       ; (gen[1] <<= 1)
    shiftlc gen[3].32B, one.32B ; (gen[3] <<= 1) + carry
    set gen[4].32B, 0           ; offset = 0

convertStr_carryloop:
    mov gen[5].32B, gen[3].32B      ; tmp = result
    shiftr gen[5].32B, gen[4].32B   ; tmp >>= offset
    bndi gen[5].32B, 0xF            ; tmp &= 0b1111
    cmpi gen[5].32B, 5              ; tmp - 5 >= 0
    jmp.neg zero.32B, convertStr_carryskip  ; if (tmp < 5) skip

    set gen[6].32B, 3               ; gen[6] = 3
    shiftl gen[6].32B, gen[4].32B   ; 3 <<= offset
    add gen[3].32B, gen[6].32B      ; result += (3 <<= offset)

convertStr_carryskip:
    addi gen[4].32B, 4              ; offset += 4
    cmpi gen[4].32B, 64             ; offset == 64
    ijmp.zero zero.32B, convertStr_carryloop    ; if (offset != 64) continue loop

    addi gen[7].32B, 1              ; counter++
    cmpi gen[7].32B, 64             ; counter == 64
    jmp.zero zero.32B, convertStr_exit  ; if (counter == 64) break

    jmp.one zero.32B, convertStr_mainloop   ; continue loop

convertStr_exit:
    set gen[7].32B, 0               ; counter = 0
    set gen[8].32B, 0               ; sum = 0

convertStr_charloop:
    mov gen[5].32B, gen[3].32B      ; tmp = result
    bndi gen[5].32B, 0xf            ; tmp &= 0b1111
    add gen[8].32B, gen[5].32B      ; sum += tmp

    cmpi gen[8].32B, 0              ; sum == 0
    jmp.zero zero.32B, convertStr_exitchar  ; if (sum == 0) skip

    addi gen[5].32B, 48             ; tmp += 48
    st.gen[2] gen[5].32B, 0         ; mem[base] = tmp

convertStr_skipchar:
    addi gen[2].32B, 4              ; base += 4s
    shiftri gen[3].32B, 4           ; result >>= 4
    addi gen[7].32B, 1              ; counter++
    cmpi gen[7].32B, 16             ; counter == 16
    ijmp.zero zero.32B, convertStr_charloop ; if (counter != 16) loop

    ; exit
    st.gen[2] zero.32B, 0           ; mem[base] = 0
    ret                             ; return


    ; <print>
    ; in: mem[strBuff] -> string
    ; use: gen[1] -> char
    ; use: gen[2] -> base
func_print: 
    set gen[2].32B, strBuff     ; base = strBuff

print_mainloop:
    ld.gen[2] gen[1].32B, 0     ; char = mem[base]
    cmpi gen[1].32B, 0          ; char == 0
    jmp.zero zero.32B, print_exit   ; if (char == 0) break

    st.gen[24] gen[1].32B, 0    ; mem[IO_BASE, TERM_OUT] = tmp
    addi gen[2].32B, 4          ; base += 4
    jmp.one zero.32B, print_mainloop    ; continue loop

print_exit:
    ret
